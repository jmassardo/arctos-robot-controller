Robot Controller's enterprise database transformation. The new schema is
designed to support **high-performance robotic operations**, **comprehensive
audit capabilities**, and **scalable multi-tenant architecture** while
maintaining **data integrity** and **optimal query performance**.

**Schema Design Principles:**

- ✅ **Third Normal Form**: Optimal data normalization with strategic
  denormalization
- ✅ **Performance-First Indexing**: Strategic index design for sub-second query

- ✅ **Enterprise Security**: Row-level security and comprehensive audit trails
- ✅ **Scalable Architecture**: Horizontal partitioning and sharding-ready
  design

### 1. Query Optimization Patterns

#### 1.1 Efficient Data Retrieval Patterns

```sql
-- Optimized user authentication query
PREPARE authenticate_user (TEXT) AS
SELECT
    u.id, u.username, u.role, u.status,
    u.password_hash, u.two_factor_enabled,
    u.failed_login_attempts, u.locked_until
FROM users u
WHERE u.username = $1 AND u.status = 'active'
LIMIT 1;

-- Optimized position listing with pagination
PREPARE list_positions (UUID, INTEGER, INTEGER) AS
WITH position_summary AS (
    SELECT
        p.id, p.name, p.description, p.status,
        p.created_at, p.updated_at, p.execution_count,
        u.username as creator_name,
        pg.name as group_name,
        pg.color as group_color
    FROM positions p
    JOIN users u ON p.created_by = u.id
    LEFT JOIN position_groups pg ON p.position_group_id = pg.id
    WHERE p.robot_config_id = $1 AND p.status != 'archived'
    ORDER BY p.is_favorite DESC, p.updated_at DESC
    LIMIT $2 OFFSET $3
)
SELECT * FROM position_summary;

-- Optimized execution history with aggregations
PREPARE execution_summary (UUID, INTERVAL) AS
SELECT
    DATE_TRUNC('day', started_at) as execution_date,
    COUNT(*) as total_executions,
    COUNT(*) FILTER (WHERE status = 'completed') as successful_executions,
    COUNT(*) FILTER (WHERE status = 'failed') as failed_executions,
    AVG(total_duration_ms) FILTER (WHERE status = 'completed') as avg_duration,
    SUM(total_duration_ms) FILTER (WHERE status = 'completed') as total_runtime
FROM execution_history
WHERE robot_config_id = $1
  AND started_at >= NOW() - $2
GROUP BY DATE_TRUNC('day', started_at)
ORDER BY execution_date DESC;
```

### 2. Database Maintenance Procedures

#### 2.1 Automated Maintenance Tasks

```sql
-- Automated table maintenance function
CREATE OR REPLACE FUNCTION maintain_database_performance()
RETURNS TEXT AS $$
DECLARE
    maintenance_log TEXT := '';
    table_record RECORD;
    index_record RECORD;
BEGIN
    -- Update table statistics
    maintenance_log := maintenance_log || 'Starting ANALYZE...' || chr(10);
    ANALYZE;
    maintenance_log := maintenance_log || 'ANALYZE completed.' || chr(10);

    -- Reindex fragmented indexes (> 20% bloat)
    FOR index_record IN
        SELECT schemaname, indexname, tablename
        FROM pg_stat_user_indexes
        WHERE idx_scan > 1000 -- Only frequently used indexes
    LOOP
        BEGIN
            EXECUTE format('REINDEX INDEX CONCURRENTLY %I.%I',
                         index_record.schemaname, index_record.indexname);
            maintenance_log := maintenance_log || format('Reindexed %s.%s' || chr(10),
                             index_record.schemaname, index_record.indexname);
        EXCEPTION WHEN OTHERS THEN
            maintenance_log := maintenance_log || format('Failed to reindex %s.%s: %s' || chr(10),
                             index_record.schemaname, index_record.indexname, SQLERRM);
        END;
    END LOOP;

    -- Clean up old partitions (older than 2 years)
    DELETE FROM audit_events
    WHERE timestamp < NOW() - INTERVAL '2 years';

    DELETE FROM execution_monitoring
    WHERE timestamp < NOW() - INTERVAL '90 days';

    maintenance_log := maintenance_log || 'Partition cleanup completed.' || chr(10);

    -- Update materialized views
    REFRESH MATERIALIZED VIEW CONCURRENTLY mv_execution_summary;
    REFRESH MATERIALIZED VIEW CONCURRENTLY mv_robot_utilization;
    maintenance_log := maintenance_log || 'Materialized views refreshed.' || chr(10);

    RETURN maintenance_log;
END;
$$ LANGUAGE plpgsql;

-- Schedule maintenance tasks
SELECT cron.schedule('database-maintenance', '0 2 * * 0', 'SELECT maintain_database_performance()');
```

#### 2.2 Performance Monitoring Views

```sql
-- Database performance monitoring views
CREATE VIEW v_database_performance AS
SELECT
    schemaname,
    tablename,
    seq_scan,
    seq_tup_read,
    idx_scan,
    idx_tup_fetch,
    n_tup_ins + n_tup_upd + n_tup_del as total_writes,
    n_live_tup,
    n_dead_tup,
    CASE
        WHEN n_dead_tup > n_live_tup * 0.1 THEN 'needs_vacuum'
        WHEN n_dead_tup > n_live_tup * 0.05 THEN 'consider_vacuum'
        ELSE 'ok'
    END as vacuum_status,
    last_vacuum,
    last_autovacuum,
    last_analyze
FROM pg_stat_user_tables
ORDER BY n_dead_tup DESC;

-- Slow query identification
CREATE VIEW v_slow_queries AS
SELECT
    query,
    calls,
    total_exec_time,
    mean_exec_time,
    rows,
    100.0 * shared_blks_hit / nullif(shared_blks_hit + shared_blks_read, 0) AS hit_percent
FROM pg_stat_statements
WHERE mean_exec_time > 100 -- Queries slower than 100ms
ORDER BY mean_exec_time DESC;

-- Index usage analysis
CREATE VIEW v_index_usage AS
SELECT
    schemaname,
    tablename,
    indexname,
    idx_scan,
    idx_tup_read,
    idx_tup_fetch,
    pg_size_pretty(pg_relation_size(indexrelid)) as index_size,
    CASE
        WHEN idx_scan = 0 THEN 'unused'
        WHEN idx_scan < 100 THEN 'rarely_used'
        WHEN idx_scan < 1000 THEN 'moderately_used'
        ELSE 'heavily_used'
    END as usage_category
FROM pg_stat_user_indexes
ORDER BY idx_scan DESC;
```
